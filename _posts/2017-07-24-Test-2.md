def mergeMaps[TKey, TValue](maps: Seq[Map[TKey, TValue]]): Map[TKey, TValue] = {
  maps.foldLeft(Map.empty[TKey, TValue]) {                                      
    (a: Map[TKey, TValue], b: Map[TKey, TValue]) =>                             
      a ++ b                                                                    
  }                                                                             
}
